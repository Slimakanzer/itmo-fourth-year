    ```
    beginp
        H(I) = 0; для ∀i = 0,2^Q - 1

        for (m = 1; m <= M; m++)
            for (n = 1; n <= M; n++)
                I = B_nm;
                H(i) += 1;
            end
        end

        D = 0;
    ```
